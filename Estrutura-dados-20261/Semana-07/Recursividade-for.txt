#include <stdio.h>

void fibonacci_iterativo(int n) {
   int a = 0, b = 1, temp;
    printf("Os primeiros %d termos da sequência de Fibonacci sao: ", n);
    for (int i = 0; i < n; i++) {
        printf("%d ", a);
        temp = a;
        a = b;
        b = temp + b;
    }
    printf("\n");
}

int main() {
    int termos = 10;
    fibonacci_iterativo(termos);
    return 0;
}
